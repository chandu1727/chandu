// Online C++ compiler to run C++ program online
#include <iostream>
using namespace std;
int main() {
   char day;
   cout <<"enter a character corresponding to the day of the week (m,t,w,h,f,s,y) :";
   cin >> day;
   //switch statement to determine the day of the week
   switch(day)
   {
          case 'm':
       cout << "monday" <<endl;
       break;
       
          case 't':
       cout << "tuesday" <<endl;
       break;
       
          case 'w':
       cout << "wednesday" <<endl;
       break;
       
          case 'h':
       cout << "thursday" <<endl;
       break;
       
          case 'f':
       cout << "friday" <<endl;
       break;
       
          case 's':
       cout << "saturday" <<endl;
       break;
       
          case 'y':
       cout << "sunday" <<endl;
       break;
       default:
          cout << "invalid input please enter a number between m and y "<<endl;
          break;
}
    return 0;
}
